Convert between the following controller integral (reset) values:

\vskip 10pt

\begin{itemize}
\item{} 34 minutes per repeat = \underbar{\hskip 50pt} repeats per minute
\vskip 5pt
\item{} 2.5 repeats per minute = \underbar{\hskip 50pt} minutes per repeat
\vskip 5pt
\item{} 100 seconds per repeat = \underbar{\hskip 50pt} repeats per minute
\vskip 5pt
\item{} 0.2 repeats per minute = \underbar{\hskip 50pt} seconds per repeat
\vskip 5pt
\item{} 0.95 repeats per minute = \underbar{\hskip 50pt} minutes per repeat
\end{itemize}

%(BEGIN_ANSWER)

\begin{itemize}
\item{} 34 minutes per repeat = \underbar{\bf 0.0294} repeats per minute
\item{} 2.5 repeats per minute = \underbar{\bf 0.4} minutes per repeat
\item{} 100 seconds per repeat = \underbar{\bf 0.6} repeats per minute
\item{} 0.2 repeats per minute = \underbar{\bf 300} seconds per repeat
\item{} 0.95 repeats per minute = \underbar{\bf 1.053} minutes per repeat
\end{itemize}
