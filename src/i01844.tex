\item{} $dy \over dx$ = -1
\vskip 10pt 
\item{} $dy \over dx$ = +2
\vskip 10pt 
\item{} $dy \over dx$ = -2
\vskip 10pt 
\item{} $dy \over dx$ = +3
\vskip 10pt 
\item{} $dy \over dx$ = -3
