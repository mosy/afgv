The {\it reset} state of a flip-flop circuit is defined as:

\begin{itemize}
\item{(A)} $Q$ = 1 and $\overline{Q}$ = 0
\vskip 5pt 
\item{(B)} $Q$ = 0 and $\overline{Q}$ = 1
\vskip 5pt 
\item{(C)} $Q$ = 1 and $\overline{Q}$ = 1
\vskip 5pt 
\item{(D)} Clock = falling edge
\vskip 5pt 
\item{(E)} $Q$ = 0 and $\overline{Q}$ = 0
\end{itemize}

%(END_QUESTION)





%(BEGIN_ANSWER)
