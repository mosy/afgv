\begin{itemize}
\item{} 86 = \underbar{\bf Lockout} or \underbar{\bf Auxiliary}
\vskip 10pt
\item{} 50 = \underbar{\bf Overcurrent (instantaneous)}
\vskip 10pt
\item{} 87 = \underbar{\bf Differential}
\vskip 10pt
\item{} 40 = \underbar{\bf Loss of excitation}
\vskip 10pt
\item{} 51 = \underbar{\bf Overcurrent (time)}
\vskip 10pt
\item{} 79 = \underbar{\bf Automatic reclose}
\vskip 10pt
\item{} 81 = \underbar{\bf Frequency}
\vskip 10pt
\item{} 52 = \underbar{\bf AC circuit breaker}
\end{itemize}
