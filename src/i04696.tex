Perform the following numeration system conversions:

\begin{itemize}
\item{} 100011 (binary) into octal: \underbar{\hskip 50pt}
\vskip 10pt
\item{} 25 (decimal) into binary: \underbar{\hskip 50pt}
\vskip 10pt
\item{} 67 (octal) into decimal: \underbar{\hskip 50pt}
\end{itemize}

\underbar{file i04696}
%(END_QUESTION)





%(BEGIN_ANSWER)

\begin{itemize}
\item{} 100011 (binary) into octal: \underbar{\bf 43}
\vskip 10pt
\item{} 25 (decimal) into binary: \underbar{\bf 11001}
\vskip 10pt
\item{} 67 (octal) into decimal: \underbar{\bf 55}
\end{itemize}
